---
title: List Diary Entries
description: Retrieve diary entries with optional filters for date ranges and entry types
---

## Endpoint

```
GET /diary
```

## Authentication

This endpoint requires authentication via Bearer token.

```bash
Authorization: Bearer <token>
```

## Query Parameters

<ParamField query="date" type="string">
  Filter entries by specific date in YYYY-MM-DD format (e.g., "2024-03-15")
</ParamField>

<ParamField query="startDate" type="string">
  Start date for date range filter in YYYY-MM-DD format
</ParamField>

<ParamField query="endDate" type="string">
  End date for date range filter in YYYY-MM-DD format
</ParamField>

<ParamField query="type" type="string">
  Filter by entry type. Options: `meal`, `symptom`, `all`
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="array">
  Array of diary entries
  
  <ResponseField name="_id" type="string">
    Entry ID (MongoDB ObjectId)
  </ResponseField>
  
  <ResponseField name="userId" type="string">
    User ID who created the entry
  </ResponseField>
  
  <ResponseField name="date" type="string">
    Entry date in YYYY-MM-DD format
  </ResponseField>
  
  <ResponseField name="type" type="string">
    Entry type: `meal` or `symptom`
  </ResponseField>
  
  <ResponseField name="meal" type="object" optional>
    Meal data (only present when type is "meal")
    
    <ResponseField name="type" type="string">
      Meal type: `breakfast`, `lunch`, `dinner`, or `snack`
    </ResponseField>
    
    <ResponseField name="time" type="string">
      Time in HH:MM format
    </ResponseField>
    
    <ResponseField name="foods" type="array">
      Array of foods consumed
      
      <ResponseField name="foodId" type="number">
        Food database ID
      </ResponseField>
      
      <ResponseField name="foodName" type="string">
        Name of the food
      </ResponseField>
      
      <ResponseField name="portion" type="string" optional>
        Portion description
      </ResponseField>
      
      <ResponseField name="quantity_g" type="number" optional>
        Quantity in grams
      </ResponseField>
      
      <ResponseField name="markedAsBad" type="boolean" optional>
        Whether this food was marked as problematic
      </ResponseField>
      
      <ResponseField name="calculatedNutrition" type="object" optional>
        Calculated nutritional values
        
        <ResponseField name="calories" type="number">Calories (kcal)</ResponseField>
        <ResponseField name="carbs" type="number">Carbohydrates (g)</ResponseField>
        <ResponseField name="protein" type="number">Protein (g)</ResponseField>
        <ResponseField name="fat" type="number">Fat (g)</ResponseField>
        <ResponseField name="sugar" type="number">Sugar (g)</ResponseField>
        <ResponseField name="fiber" type="number">Fiber (g)</ResponseField>
        <ResponseField name="sodium" type="number">Sodium (mg)</ResponseField>
      </ResponseField>
    </ResponseField>
    
    <ResponseField name="notes" type="string" optional>
      Additional notes about the meal
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="symptom" type="object" optional>
    Symptom data (only present when type is "symptom")
    
    <ResponseField name="type" type="string">
      Symptom type: `bloating`, `gas`, `cramps`, `nausea`, `diarrhea`, `constipation`, `reflux`, `fatigue`, `headache`, `brain_fog`, or `other`
    </ResponseField>
    
    <ResponseField name="intensity" type="number">
      Intensity level from 1 (mild) to 5 (severe)
    </ResponseField>
    
    <ResponseField name="time" type="string">
      Time in HH:MM format
    </ResponseField>
    
    <ResponseField name="duration" type="number" optional>
      Duration in minutes
    </ResponseField>
    
    <ResponseField name="notes" type="string" optional>
      Additional notes about the symptom
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="createdAt" type="string">
    Timestamp when the entry was created
  </ResponseField>
  
  <ResponseField name="updatedAt" type="string">
    Timestamp when the entry was last updated
  </ResponseField>
</ResponseField>

## Examples

### Get all entries for a specific date

```bash
curl -X GET "https://api.ceboelha.com/diary?date=2024-03-15" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

### Get all meal entries in a date range

```bash
curl -X GET "https://api.ceboelha.com/diary?startDate=2024-03-01&endDate=2024-03-31&type=meal" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

### Get only symptom entries

```bash
curl -X GET "https://api.ceboelha.com/diary?type=symptom" \
  -H "Authorization: Bearer YOUR_TOKEN"
```
