title: Get Profile
description: Retrieve the authenticated user's profile information

## Endpoint

```
GET /profile
```

Retrieves the complete profile information for the currently authenticated user, including personal details, preferences, diet settings, and usage statistics.

## Authentication

This endpoint requires authentication. Include a valid JWT token in the Authorization header.

## Request

No request body or parameters required.

### Headers

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
  
  Example: `Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...`
</ParamField>

## Response

<ResponseField name="success" type="boolean" required>
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object" required>
  User profile object
  
  <ResponseField name="id" type="string" required>
    Unique user identifier
  </ResponseField>
  
  <ResponseField name="email" type="string" required>
    User's email address
  </ResponseField>
  
  <ResponseField name="name" type="string" required>
    User's display name
  </ResponseField>
  
  <ResponseField name="avatar" type="string">
    URL to user's profile picture
  </ResponseField>
  
  <ResponseField name="role" type="string" required>
    User role: `user` or `admin`
  </ResponseField>
  
  <ResponseField name="status" type="string" required>
    Account status: `active`, `inactive`, or `banned`
  </ResponseField>
  
  <ResponseField name="preferences" type="object" required>
    User preferences and settings
    
    <ResponseField name="theme" type="string">
      UI theme: `light`, `dark`, or `system`
    </ResponseField>
    
    <ResponseField name="notifications" type="boolean">
      Whether notifications are enabled
    </ResponseField>
    
    <ResponseField name="soundEnabled" type="boolean">
      Whether sound effects are enabled
    </ResponseField>
    
    <ResponseField name="language" type="string">
      Interface language: `pt-BR` or `en`
    </ResponseField>
    
    <ResponseField name="fodmapPhase" type="string">
      Current FODMAP diet phase: `elimination`, `reintroduction`, or `personalization`
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="dietSettings" type="object" required>
    Diet tracking configuration
    
    <ResponseField name="enabled" type="boolean">
      Whether diet tracking is enabled
    </ResponseField>
    
    <ResponseField name="preset" type="string">
      Diet preset: `custom`, `maintenance`, `cutting`, `bulking`, `lowcarb`, or `balanced`
    </ResponseField>
    
    <ResponseField name="limits" type="object">
      Daily nutritional limits
      
      <ResponseField name="calories" type="number">
        Daily calorie limit
      </ResponseField>
      
      <ResponseField name="carbs" type="number">
        Daily carbohydrates limit (grams)
      </ResponseField>
      
      <ResponseField name="protein" type="number">
        Daily protein limit (grams)
      </ResponseField>
      
      <ResponseField name="fat" type="number">
        Daily fat limit (grams)
      </ResponseField>
      
      <ResponseField name="sugar" type="number">
        Daily sugar limit (grams)
      </ResponseField>
      
      <ResponseField name="fiber" type="number">
        Daily fiber limit (grams)
      </ResponseField>
      
      <ResponseField name="sodium" type="number">
        Daily sodium limit (mg)
      </ResponseField>
    </ResponseField>
    
    <ResponseField name="showRemaining" type="boolean">
      Show remaining nutrients instead of consumed
    </ResponseField>
    
    <ResponseField name="showProgressBars" type="boolean">
      Display progress bars in diary
    </ResponseField>
    
    <ResponseField name="warningThreshold" type="number">
      Percentage threshold for warnings (0-100)
    </ResponseField>
    
    <ResponseField name="diaryMode" type="string">
      Diary display mode: `quick` or `detailed`
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="stats" type="object" required>
    User activity statistics
    
    <ResponseField name="daysUsingApp" type="number">
      Total days using the app
    </ResponseField>
    
    <ResponseField name="totalMealsLogged" type="number">
      Total number of meals logged
    </ResponseField>
    
    <ResponseField name="totalSymptomsLogged" type="number">
      Total number of symptoms logged
    </ResponseField>
    
    <ResponseField name="currentStreak" type="number">
      Current consecutive days streak
    </ResponseField>
    
    <ResponseField name="longestStreak" type="number">
      Longest consecutive days streak
    </ResponseField>
    
    <ResponseField name="achievementsUnlocked" type="number">
      Number of achievements unlocked
    </ResponseField>
    
    <ResponseField name="foodsTested" type="number">
      Number of foods tested
    </ResponseField>
    
    <ResponseField name="triggersIdentified" type="number">
      Number of trigger foods identified
    </ResponseField>
    
    <ResponseField name="lastActive" type="string">
      ISO 8601 timestamp of last activity
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="createdAt" type="string" required>
    ISO 8601 timestamp of account creation
  </ResponseField>
  
  <ResponseField name="updatedAt" type="string" required>
    ISO 8601 timestamp of last profile update
  </ResponseField>
</ResponseField>

## Example

```bash
curl -X GET https://api.ceboelha.com/profile \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN"
```

## Error Responses

### 401 Unauthorized

```json
{
  "success": false,
  "error": "Token inválido ou expirado"
}
```

### 404 Not Found

```json
{
  "success": false,
  "error": "Usuário não encontrado"
}
```
